\begin{frame}{Research Design Overview}

\textbf{Question}: Is the sector-level allocation of BNDES lending across municipalities GDP-optimal?

\medskip

\textbf{Three-step identification}:
\begin{enumerate}
    \item \textbf{Firm first stage} (micro validation): political alignment $\to$ firm BNDES access
    \item \textbf{Sector aggregation}: firm instruments $\to$ sector-level credit reallocation
    \item \textbf{Municipality second stage}: predicted reallocation $\to$ GDP per capita
\end{enumerate}

\medskip

\textbf{This presentation}: steps 1 and 2 --- establishing (and diagnosing) the first stage

\smallskip

\begin{itemize}
    \item Firm-level signal is \textbf{present}: alignment predicts BNDES access
    \item Sector-level aggregation \textbf{destroys} the signal --- we diagnose why
    \item Forward paths: coarser sectors, firm-level reallocation summaries, ungrouped CNAE
\end{itemize}

\end{frame}

% ===========================================================================
\begin{frame}[shrink=5]{Firm-Level Specification}

\textbf{Unit}: firm $f$ $\times$ municipality $m$ $\times$ year $t$ \quad ($N = 39.5$M, 2002--2017)

\smallskip

\textbf{Dependent variable}: $\mathbf{1}(\text{BNDES}_{fmt} > 0)$ --- did firm $f$ receive any BNDES credit in muni $m$, year $t$? (extensive margin, LPM)

\smallskip

\textbf{Instruments} --- ``Fraction Aligned'' at government tier $\ell \in \{\text{Mayor, Governor, President}\}$:
\begin{equation*}
    \FA^\ell_{fmt} \;=\; \sum_{p} \omega_{fp,0} \;\cdot\; \Align^\ell_{mpt}
\end{equation*}

\vspace{-6pt}
\begin{itemize}\setlength{\itemsep}{2pt}
    \item $\omega_{fp,0}$: share of firm $f$'s owners affiliated with party $p$, averaged over the 4-year pre-election window (baseline exposure)
    \item $\Align^\ell_{mpt}$: $=1$ if party $p$ is in the governing coalition at tier $\ell$ in muni $m$, year $t$
    \item $\FA^\ell$: fraction of the firm's politically affiliated owners whose party is currently in power at tier $\ell$
\end{itemize}

\smallskip

\textbf{Specification}:
\begin{equation*}
    \mathbf{1}(\text{BNDES}_{fmt} > 0) = \sum_\ell \beta_\ell \, \FA^\ell_{fmt} + \phi_f + \alpha_{mt} + \varepsilon_{fmt}
\end{equation*}

\vspace{-6pt}
{\footnotesize FE: firm + muni$\times$year. Clustering: firm + muni. Weights (optional): employment.}

\end{frame}

% ===========================================================================
\begin{frame}{Aggregated Firm $\to$ Sector Specification}

\textbf{Collapse} firm panel to \textbf{(sector group $j$, municipality $m$, year $t$)} cells:

\smallskip

\textbf{Outcome}: share of pre-election base firms with any BNDES
\begin{equation*}
    H^{\mathrm{pre}}_{jmt} = \frac{1}{N^{\mathrm{pre}}_{jme}} \sum_{f \in \mathcal{F}^{\mathrm{pre}}_{jme}} \mathbf{1}(\text{BNDES}_{fmt} > 0)
\end{equation*}

\vspace{-6pt}
\textbf{Instruments}: simple average of firm-level $\FA$ within each cell
\begin{equation*}
    \overline{\FA}^\ell_{jmt} = \frac{1}{N^{\mathrm{pre}}_{jme}} \sum_{f \in \mathcal{F}^{\mathrm{pre}}_{jme}} \FA^\ell_{fmt}
\end{equation*}

\vspace{-6pt}
\textbf{Specification}:
\begin{equation*}
    H^{\mathrm{pre}}_{jmt} = \sum_\ell \lambda_\ell\, \overline{\FA}^\ell_{jmt} + \phi_{jm} + \alpha_{mt} + u_{jmt}
\end{equation*}

\vspace{-4pt}
{\footnotesize FE: muni$\times$sector + muni$\times$year. Clustering: muni + sector. Pre-election base $\mathcal{F}^{\mathrm{pre}}_{jme}$: firms with any non-zero $\FA$ instrument. This is the \emph{same} political signal as the firm spec, averaged within cells.}

\end{frame}

\end{enumerate}

\begin{frame}[shrink=5]{Sector First Stage: Key Findings Across Specifications}

\begin{itemize}\setlength{\itemsep}{4pt}
    \item \textbf{Equal-firm, coalition, j$\times$t FE}: best single-combo $F = 10.7$ --- the only specification exceeding Stock-Yogo 10
    \item \textbf{Employment-weighted}: second strongest ($F_{\max} = 8.3$); approaching relevance
    \item \textbf{Owner-count}: weak under coalition ($F \leq 5.9$), near-zero under party alignment
    \item \textbf{Binary}: weakest weight variant at sector level ($F \leq 3.4$) --- opposite of firm-level pattern
    \item \textbf{Party alignment uniformly weak}: $F < 3$ across all weight variants and FE choices
    \item \textbf{m$\times$t FE}: reduces all $F$ relative to j$\times$t FE (absorbs muni-year variation)
\end{itemize}

\bigskip

\textbf{Interpretation}: Equal-firm and employment weights concentrate exposure on municipalities where BNDES lending actually varies across sectors. Coalition alignment is the operative margin at the sector level. The sector pipeline shows \textbf{some promise under specific weight/FE combinations}, unlike the uniformly weak aggregated firm$\to$sector route.

\end{frame}

\begin{frame}[shrink=8]{Alternative Paths Forward}

\textbf{Option A: Coarser sector groupings aligned with BNDES lending}
\begin{itemize}\setlength{\itemsep}{1pt}
    \item Define 3--5 ``BNDES-relevant'' sectors based on where lending occurs and affiliation support exists
    \item Evidence: Transport and Residual carry the signal --- perhaps a Manufacturing / Services / Transport+Other trichotomy
    \item \textcolor{notegreen}{Pro}: denser cells, more outcome variation, preserves sector story
    \item \textcolor{alertred}{Con}: fewer sectors $\Rightarrow$ less heterogeneity for second stage
\end{itemize}

\smallskip

\textbf{Option B: Firm-level reallocation summary variable}
\begin{itemize}\setlength{\itemsep}{1pt}
    \item Construct a municipality-level measure of \emph{firm-level} BNDES reallocation (e.g., turnover in firm access between $t{-}1$ and $t$, or firm-level HHI within municipality)
    \item Instrument with municipality-level alignment --- \textbf{bypass} the sector dimension
    \item \textcolor{notegreen}{Pro}: uses the strong firm-level signal directly; no sector-aggregation noise
    \item \textcolor{alertred}{Con}: new theoretical motivation needed; less interpretable for sector misallocation
\end{itemize}

\smallskip

\textbf{Option C: Return to ungrouped sectors (21 CNAE sections)}
\begin{itemize}\setlength{\itemsep}{1pt}
    \item Ungrouped with muni$\times$year FE achieves $F = 12.4$ for mayor (above Stock-Yogo 10)
    \item More sections $\to$ more variation in shares and instruments
\end{itemize}

\end{frame}

\begin{frame}{Conclusion \& Next Steps}

\textbf{Firm-level validation is solid}: political alignment $\to$ BNDES access is a real channel.

\bigskip

\textbf{Binary exposure is the key finding}:
\begin{itemize}
    \item \textbf{Firm level}: binary president $F = 7.4$ vs.\ pooled-count governor $F = 6.9$
    \item \textbf{Aggregated}: binary survives ($F$ up to 18.8) while pooled-count collapses ($F < 2.3$)
    \item \textbf{Sector shift-share}: equal-firm weights reach $F = 10.7$; employment $F = 8.3$
\end{itemize}

\bigskip

\textbf{Remaining challenges}: 53\% of cells have $\leq 2$ firms; 63\% of muni-years have $\leq 1$ positive sector; party alignment uniformly weak at sector level.

\bigskip

\textbf{Next steps}:
\begin{enumerate}
    \item Pursue \textbf{equal-firm / employment sector instruments} with j$\times$t FE as the primary sector specification
    \item Investigate the \textbf{binary aggregated route} (binary party, emp-weighted, $F = 18.8$) as an alternative pathway
    \item Run ungrouped sector first stage (21 CNAE sections) --- earlier result suggested $F = 12.4$ for mayor
    \item Evaluate evidence and choose the path to the second stage (GDP)
\end{enumerate}

\end{frame}

\begin{itemize}\setlength{\itemsep}{1pt}

\end{itemize}

\begin{itemize}\setlength{\itemsep}{1pt}
    \item FE: muni $\times$ sector group + muni $\times$ year (8,606 FE groups; 2,049 singletons removed)
    \item Two-way clustering: municipality + sector group
    \item Effective sample: 292,926 observations after singleton removal
\end{itemize}

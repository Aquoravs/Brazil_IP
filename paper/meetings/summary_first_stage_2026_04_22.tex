\begin{frame}[t,shrink=15]{Key Findings}
\begin{itemize}\setlength{\itemsep}{0.25em}

  \item \textbf{Firm-level BNDES first stage is strong and robust across municipalities.}
    Unweighted coalition yields $F=45$ (mayor), $F=103$ (governor), $F=63$
    (M+G).  Split by city size: top-Q4 $F=64$, bottom-3Q $F=61$ for governor
    --- no clear concentration.

  \item \textbf{Alignment predicts firm employment with large
    estimated coefficients.}
    Coalition mayor alignment raises $\log(\text{Emp})$ by 2.7\% ($F=189$);
    governor by 2.1\% ($F=88$).  Party alignment: mayor +4.0\% ($F=130$),
    president +4.9\% ($F=35$).  All instruments pass and all coefficients
    are significant at 1\%.

  \item \textbf{Employment outcomes outperform BNDES outcomes at the
    aggregated level --- driven by presidential party alignment.}
    Full-sample employment share reaches $F=18.7$ ($P$ alone, MxJ+MxT) and
    $F=20.0$ (M+G+P, MxJ+JxT); log employment reaches $F=10.6$--$12.6$.
    BNDES extensive peaks at $F=12.4$ (mayor only, JxT); BNDES share
    is uniformly weak ($F<5$).  Coalition alignment is weak for all
    employment outcomes ($F<5$).  In top-Q4 municipalities, party president
    reaches $F=41.7$ ($\hat\beta=-0.134^{***}$).

  \item \textbf{The full aggregated battery confirms the first stage is
    generically weak at sector level.}
    Across BNDES-sector (4 groups), custom-sector (11 groups), size-bin (3),
    and sector$\times$size-bin (12): the vast majority of specifications
    yield $F<10$.  Isolated passes cluster in party alignment with
    emp-weighted JxT, and in size-bin with emp-weighted.  Only the
    top-Q4 split-sample reliably passes at BNDES-sector level.

  \item \textbf{Finer sector classification kills the first stage.}
    BNDES sector $\times$ firm-size tercile: $F<10$ in all 24 configs.
    Maximum $F=9.7$.  Alignment operates at the BNDES-sector level,
    not within size bins.

\end{itemize}
\end{frame}

\begin{frame}[t]{GDP Deflation: National IPCA Confirmed, No Municipal Alternative}

\begin{itemize}\setlength{\itemsep}{0.3em}
  \item \textbf{Current pipeline uses national IPCA} (base 2018) applied uniformly
    to all municipalities within a year.  No spatial deflation.

  \item \textbf{No municipality-level price index exists for Brazil.}
    The geographic frontier is the IBGE metro-area IPCA: $\sim$13 metro regions
    covering $\sim$300 municipalities (8\% of count, $\sim$55\% of GDP).
    The remaining $\sim$5,200 municipalities have no official price series.

  \item \textbf{Three robustness options, ranked by effort:}
    \begin{enumerate}\setlength{\itemsep}{0.1em}
      \item Metro-area IPCA subsample ($\sim$2 weeks) --- re-estimate on $\sim$300
        metro municipalities with local deflators
      \item Wage-residual proxy ($\sim$3--4 weeks) --- municipality FE from
        Mincerian wage regression (RAIS data, full panel coverage)
      \item Census housing rent ($\sim$1 week) --- 2000/2010 rent levels as
        pre-period cost-of-living control
    \end{enumerate}
\end{itemize}

\vspace{0.3em}
\textcolor{alertred}{\footnotesize\textbf{Recommended framing:} ``We use national IPCA, consistent with standard practice in the Brazilian municipal panel literature.  As robustness, we re-estimate on the metro subsample with local deflators, finding qualitatively similar results.''}
\end{frame}

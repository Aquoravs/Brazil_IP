\begin{frame}{Tasks done}
\begin{itemize}\setlength{\itemsep}{0.9em}
\item \textbf{Which political office moves local economic composition?}
      Instead of using Mayor-crossed channels by assumption, we can verify which of them actually move sector shares.
\item \textbf{Stacking different offices and their interactions can produce better results?}
      Whether to test the interaction in isolation or stacked with their constituent channels.
\item \textbf{Updated version of the AR test.} The AR test now includes an instrumented version of volume controls and is also evaluated for sector by size groups. 
\end{itemize}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}{Summary}
\begin{itemize}\setlength{\itemsep}{0.8em}
\item \textbf{First-stage answers.} At policy block, the President is the only composition-relevant channel; Governor and Mayor\,$\cdot$\,Governor are volume-relevant. At 12 groups no channel clears the composition gate, so I stick to the Mayor-crossed channels.
\item \textbf{Stacking is the better default.} \{M, G, M\,$\cdot$\,G\} stacked rejects more sharply than M\,$\cdot$\,G alone at both margins.
\item \textbf{Mayor\,$\cdot$\,Governor is the robust composition channel.} It rejects at policy block across all three volume treatments; at 12 groups it rejects without and with a volume control but not under Full IV.
\item \textbf{Open question.} Why does Full IV overturn the 12-group rejection --- weak volume instrument, or a genuine volume-composition entanglement at the finer margin?
\end{itemize}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}{We let the data decide each channel's role}
\begin{itemize}\setlength{\itemsep}{0.55em}
\item \textbf{Seven cross-office channels.} Mayor (M), Governor (G), President (P), and the pairs and triple M\,$\cdot$\,G, M\,$\cdot$\,P, G\,$\cdot$\,P, M\,$\cdot$\,G\,$\cdot$\,P.
\item \textbf{Two saturated first stages.} A \emph{composition} first stage (does the channel move sector employment shares?) and a \emph{volume} first stage (does it move total BNDES volume?).
\item \textbf{Two margins.} Policy block (4 groups: Agriculture, Industry, Infrastructure, Services) and policy block $\times$ firm size (12 groups).
\item \textbf{Three volume treatments.} No volume control; volume as a predetermined control; volume instrumented (Full IV).
\end{itemize}
\vskip 0.4em
{\footnotesize Sample: 5{,}544 municipalities; 88{,}694 municipality-years. Muni and year fixed effects; SE clustered by municipality.}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}{The routing rule, in one line}
\vskip 0.6em
\begin{itemize}\setlength{\itemsep}{0.7em}
\item \textbf{Composition gate.} A channel is composition-relevant if its partial $F$ in the composition first stage clears the 5\% threshold ($p<0.05$).
\item \textbf{Volume gate.} A channel is volume-relevant if its partial $F$ in the volume first stage clears the 5\% threshold ($p<0.05$).
\item \textbf{Assignment.} Composition-relevant $\rightarrow$ \emph{composition} instrument set. Volume-relevant only $\rightarrow$ \emph{volume} instrument. Neither $\rightarrow$ \emph{dropped}.
\item \textbf{Fallback.} If no channel clears the composition gate, the test reverts to the four Mayor-crossed channels \{M, M\,$\cdot$\,P, M\,$\cdot$\,G, M\,$\cdot$\,G\,$\cdot$\,P\} as the composition set.
\end{itemize}
\vskip 0.5em
{\footnotesize The exposure control (EC) is an included control in every specification, first stages and AR test alike.}
\end{frame}

\begin{frame}{Twelve-group thin cells carry little GDP}
\begin{itemize}\setlength{\itemsep}{0.7em}
\item At the 12-group margin, thin cells (5 or fewer affiliated owners) are \textbf{64.8\% of cells}.
\item But thin-identified municipality-years carry only \textbf{9.3\% of municipal GDP}.
\item The muni-relative weighting denominator keeps per-channel weights thick; the effective number of shocks has a median of 12.5.
\item Thin cells concentrate in small municipalities, so the GDP-relevant AR estimand is not driven by them.
\end{itemize}
\vskip 0.5em
{\footnotesize Source: C3 thin-cell coverage audit, policy block $\times$ firm size margin, 452{,}842 cell-years.}

\begin{frame}{Research Question}

\textbf{Is the sector-level allocation of BNDES lending across municipalities GDP-optimal?}

\bigskip

\begin{itemize}
    \item BNDES disburses $\sim$3\% of GDP annually in subsidized industrial lending
    \item Existing literature (Lane 2025, Juh\'asz 2018) shows IP ``works'' for targeted sectors
    \item \textbf{But}: does the policymaker target the \emph{right} sectors?
    \item We derive and test the \textbf{first-order condition for optimality}: at an interior optimum, marginal reallocations across sectors have zero effect on GDP
\end{itemize}

\bigskip

\textbf{Contribution}: Move from ``does IP work?'' to ``is IP \emph{optimal}?''
--- and if not, identify which sectors are over/under-supported

\end{frame}

% ===========================================================================
\begin{frame}{Theory: Local Optimality Test}

Sector shares $s_m \in \Delta^{J-1}$, municipal output $Y_m = Y_m(s_m)$.

\medskip

\textbf{Directional derivative condition} --- at any interior optimum:
\begin{equation*}
    \frac{dY_m}{d\varepsilon}\bigg|_{\varepsilon=0}
    = \nabla_{s_m} Y_m \cdot v_m = 0
    \quad\text{for all } v_m \perp \iotaVec
\end{equation*}

\textbf{Empirical projection} (within-simplex, $\iotaVec'\Delta s_{mt}=0$):
\begin{equation*}
    \Delta Y_{m,t+h} = \beta' \Delta s_{mt} + u_{m,t+h}
\end{equation*}

\textbf{Null hypothesis (optimality)}:
\begin{equation*}
    H_0: \quad \beta = \mathbf{0}
\end{equation*}

\begin{itemize}
    \item Model-free: valid under any smooth objective
    \item $\hat\beta_j > 0$: sector $j$ under-supported $\implies$ tilt more toward $j$
    \item $\hat\beta_j < 0$: sector $j$ over-supported
\end{itemize}

\end{frame}

% ===========================================================================
\begin{frame}{Definitions: Firm-Level Instrument}

\textbf{Baseline party exposure} of firm $f$ to party $p$ (4-year pre-election average):
\begin{equation*}
    \omega_{fp,\tau} \equiv \frac{1}{|T_\tau|}\sum_{s \in T_\tau} \frac{L_{f,p,s}}{L_{f,s}},
    \qquad \sum_p \omega_{fp,\tau} \leq 1
\end{equation*}
where $L_{fp}$ = owners affiliated to party $p$; $L_f$ = total owners (affiliated + unaffiliated).

\smallskip

\textbf{Political alignment}: $\Align^\ell_{mpt} = 1$ if party $p$ holds office $\ell \in \{$mayor, governor, president$\}$ in the jurisdiction of municipality $m$ at time $t$. Spread across the full 4-year term.

\smallskip

\textbf{Firm-level levels instrument} (shift $\times$ share):
\begin{equation*}
    \FA^\ell_{fmt} \equiv \sum_p \omega_{fp,\tau} \cdot \Align^\ell_{mpt}
\end{equation*}

\textbf{Firm-level changes instrument} (turnover only, not spread):
\begin{equation*}
    \Delta\FA^\ell_{fmt} \equiv \sum_p \omega_{fp,\tau} \cdot \Delta\Align^\ell_{mpt}
\end{equation*}

\smallskip

{\footnotesize The sector-level instrument $Z^\ell_{jmt} = \sum_p w_{jmp,\tau} \Align^\ell_{mpt}$ is the owner-count-weighted aggregation of $\FA^\ell_{fmt}$ within each $(j,m)$ cell. See \hyperlink{slide:aggregation}{\textcolor{darkblue}{Aggregation slide}} for details.}

\end{frame}

% ===========================================================================
\begin{frame}{Estimation Strategy: Four Margins}

\textbf{Firm first stage} (firm $\times$ municipality $\times$ year):
\begin{equation*}
    \mathbf{1}(\text{BNDES}_{fmt} > 0) = \sum_\ell \lambda_\ell\, \FA^\ell_{fmt}
    + \gamma_f + \alpha_{mt} + u_{fmt}
\end{equation*}

\vspace{-6pt}
{\small FE: firm + muni$\times$year. Clustering: firm + muni. Employment-weighted (main), unweighted (robustness).}

\medskip

\textbf{Four margins} (levels $\times$ extensive/intensive; changes $\times$ extensive/intensive):

\smallskip

\begin{center}
\footnotesize
\begin{tabular}{l >{\centering\arraybackslash}p{5.5cm} >{\centering\arraybackslash}p{5.5cm}}

\end{center}

\medskip

Nine instrument combos per margin: M, G, P, M+G, M+P, M+G+P, plus three interaction specs (M+G+MxG, M+G+P+MxG, M+G+P+MxP).

\end{frame}

\begin{frame}{Interpreting the President's Negative Sign}

\textbf{Puzzle}: Presidential coalition alignment $\to$ \emph{lower} probability of firm BNDES access.

\medskip

\textbf{Possible explanations}:
\begin{enumerate}
    \item \textbf{Crowd-out}: presidential alignment captures national-level credit that \emph{substitutes} for BNDES indirect lending (direct BNDES lines, Exim, cards)
    \item \textbf{Reallocation}: aligned presidents channel BNDES through direct operations rather than indirect (bank-intermediated) lending in our data
    \item \textbf{Compositional}: aligned firms tend to be in sectors already served by other credit channels
    \item \textbf{Measurement}: presidential alignment varies only nationally --- muni$\times$year FE absorbs the aggregate effect, leaving only cross-firm residual variation
\end{enumerate}

\medskip

\textcolor{notegreen}{\textbf{Implication for sector instruments}}: if presidential alignment operates through channels outside indirect BNDES, the president-tier instrument may violate exclusion --- motivates using \textbf{mayor + governor only} in the sector first stage.

\end{frame}

\begin{frame}[allowframebreaks]{Key Takeaways}

\textbf{Extensive margin confirmed as the active channel}:
\begin{itemize}
    \item Political alignment predicts whether a firm gets BNDES credit, not how much
    \item Effect is robust across levels and changes specifications
    \item Intensive margin shows zero signal (neither levels nor changes)
\end{itemize}

\medskip

\textbf{Which tiers matter}:
\begin{itemize}
    \item \textbf{Governor}: most robust across weighted specifications (significant in 4/5 levels specs)
    \item \textbf{President}: negative sign, dominant in changes and 2002-fixed baselines
    \item \textbf{Mayor}: weakest overall (marginally significant in main, drops in single-muni and changes)
    \item Coalition alignment outperforms party alignment for joint instrument strength
\end{itemize}

\framebreak

\textbf{Weighting matters}:
\begin{itemize}
    \item Unweighted $F$-stats are 10--20$\times$ larger --- alignment channel is strongest among small firms
    \item Employment-weighted results are economically more relevant but statistically noisier
\end{itemize}

\medskip

\textbf{Interaction instruments}:
\begin{itemize}
    \item MxG has opposite signs across margins ($+$ extensive, $-$ intensive): joint alignment spreads credit across more firms at lower amounts
    \item MxP turnover interaction partially offsets the negative president single-tier effect
    \item Interactions do not improve overall instrument strength ($F < 10$)
\end{itemize}

\end{frame}

\begin{frame}[allowframebreaks]{Firm $\to$ Sector Aggregation: Three Key Differences}
\hypertarget{slide:aggregation}{}

Start from the unweighted firm extensive margin:
\vspace{-6pt}
\[
  \mathbf{1}(\text{BNDES}_{fmt} > 0)
  = \sum_{\ell} \lambda^{\text{ext}}_{\ell}\, \FA^{\ell}_{fmt}
  + \gamma_{f} + \alpha_{mt} + u_{fmt}.
\]
\vspace{-6pt}
Average over pre-election firm base $\mathcal{F}^{\mathrm{pre}}_{jme}$:
\vspace{-6pt}
\[
  \underbrace{H^{\mathrm{pre}}_{jmt}}_{\text{share of pre-election firms w/ BNDES}}
  = \sum_{\ell} \lambda^{\text{ext}}_{\ell}\,
  \underbrace{\overline{\FA}^{\ell,\mathrm{pre}}_{jmt}}_{\text{avg firm instrument}}
  + \bar{\gamma}_{jme} + \alpha_{mt} + \bar{u}_{jmt}.
\]
\vspace{-6pt}
Compare with the sector first stage:
\vspace{-6pt}
\[
  \underbrace{s_{jmt}}_{\text{BNDES value share}} = \sum_{\ell} \lambda_{\ell}\, Z^{\ell}_{jmt}
  + \phi\,\EC_{jm,\tau} + \gamma_{jm} + \alpha_{jt} + u_{jmt}.
\]

\framebreak

\textbf{Difference 1: Outcome object.}
\begin{itemize}
    \item Aggregated firm: $H^{\mathrm{pre}}_{jmt}$ = \textbf{head-count} share of pre-election firms receiving any credit (pure extensive margin)
    \item Sector: $s_{jmt}$ = \textbf{value} share of BNDES credit flowing to sector $j$
    \item Two sub-differences:
    \begin{enumerate}
        \item \textit{Firm base}: pre-election firms (fixed) vs.\ current-year firms (time-varying)
        \item \textit{Margin}: pure extensive vs.\ extensive $+$ intensive ($s_{jmt}$ reflects how many firms \emph{and} how much each receives)
    \end{enumerate}
    \item Since firm results show zero intensive-margin signal, the mixing in $s_{jmt}$ may \textbf{not} attenuate much --- but many zeros add noise
\end{itemize}

\framebreak

\textbf{Difference 2: Instrument weighting.}
\begin{itemize}
    \item Aggregated firm weights each firm equally: $\bar{\omega}_{jmp,\tau} = \frac{1}{N^{\mathrm{pre}}_{jme}} \sum_f \omega_{fp,\tau}$
    \item Sector weights by baseline owner count: $w_{jmp,\tau} = \sum_f \frac{L_{f,\tau}}{N_{jm,\tau}} \omega_{fp,\tau}$
    \item The two differ when party exposure $\omega_{fp,\tau}$ \textbf{covaries with firm size} $L_{f,\tau}$
\end{itemize}

\medskip

\textbf{Difference 3: Fixed effects.}
\begin{itemize}
    \item Aggregated firm: $\bar{\gamma}_{jme}$ (cell avg of firm FE) $+$ muni$\times$year
    \item Sector: muni$\times$sector FE $+$ sector$\times$year FE $+$ exposure control $\EC_{jm,\tau}$
    \item Firm spec has \emph{finer} municipality conditioning but \emph{coarser} sector conditioning
\end{itemize}

\end{frame}

% ===========================================================================
\begin{frame}{Why Might the Firm Signal Not Translate to the Sector Level?}

\textbf{H1: Scale effect + cross-sector cancellation} \textcolor{notegreen}{(likely)}
\begin{itemize}
    \item Connected owners are spread across many sectors $\Rightarrow$ alignment raises BNDES in multiple sectors simultaneously
    \item Firm regression detects this level effect, but $\sum_j \Delta s_{jmt} = 0$ by construction $\Rightarrow$ gains cancel in shares
    \item Sector first stage only identifies \textbf{cross-sector reallocation}, not overall expansion
\end{itemize}

\medskip

\textbf{H2: Jensen's inequality in instrument aggregation}
\begin{itemize}
    \item Firm: $\FA_{fmt} = \sum_p (L_{fp}/L_f) \cdot \Align_{mpt}$ --- sum of \textbf{ratios}; sector: $Z_{jmt}$ uses $\sum_f L_{fp}/\sum_f L_f$ --- \textbf{ratio of sums}
    \item Small firms with high party share are overweighted in firm aggregation vs.\ sector instrument
\end{itemize}

\medskip

\textbf{H3: Thin sector coverage within municipalities}
\begin{itemize}
    \item If few sectors have positive BNDES per muni-year, $s_{jmt}$ is mostly zeros
    \item Cross-sector variation in $Z_{jmt}$ may be small, leaving little identifying variation after sector$\times$year FE
\end{itemize}

\end{frame}

% ===========================================================================
\begin{frame}{Diagnostic Agenda: Firm-to-Sector Link}

\textbf{Diagnostic 1: Cross-sector distribution of aligned firms}
\begin{itemize}
    \item Are newly aligned firms concentrated in specific sectors or spread evenly across all sectors within a municipality?
    \item Compute Herfindahl of $L_{jmp}$ across sectors $j$ for each $(m, p)$: high HHI $\Rightarrow$ alignment shocks are sector-specific (good for identification); low HHI $\Rightarrow$ shocks are diffuse (shares won't move)
\end{itemize}

\bigskip

\textbf{Diagnostic 2: Sector coverage within municipalities}
\begin{itemize}
    \item What share of the 10 sector groups has positive BNDES credit in a given municipality-year?
    \item Distribution across municipalities and evolution over time
    \item If most municipality-years have credit in only 2--3 sectors, $s_{jmt}$ has very limited cross-sector variation
\end{itemize}

\bigskip

\textbf{Diagnostic 3: Instrument variation decomposition}
\begin{itemize}
    \item Within-muni-year variance of $Z_{jmt}$ across sectors (identifying variation under sector$\times$year FE)
    \item Between-muni-year variance (absorbed by muni$\times$year FE)
    \item If within-muni-year variation is small $\Rightarrow$ sector$\times$year FE absorbs most signal
\end{itemize}

\end{frame}

% ===========================================================================
\begin{frame}{Implications for Sector First Stage}

\textbf{Firm-level findings guide the sector instrument design}:

\medskip

\begin{enumerate}
    \item \textbf{Extensive margin drives sector shares}: alignment shifts \emph{which firms} get BNDES, aggregating to shifts in $s_{jmt}$

    \item \textbf{Governor + mayor preferred}: both show positive extensive-margin effects; president's negative sign and crowd-out concern suggest excluding it

    \item \textbf{Coalition preferred over party}: stronger joint $F$, governor significant under coalition but not party

    \item \textbf{Interactions add nuance but not strength}: joint $F$-stats do not improve over single-tier specs

    \item \textbf{Cycle-specific baselines preferred}: 2002-fixed shifts which tier dominates, suggesting baseline recency matters
\end{enumerate}

\end{frame}

\textbf{Immediate --- diagnostics}:
\begin{enumerate}
    \item Sector coverage: what \% of sector groups have positive BNDES per muni-year?
    \item Alignment concentration: Herfindahl of $L_{jmp}$ across sectors --- are aligned firms sector-specific or diffuse?
    \item Instrument variation decomposition: within- vs.\ between-muni-year variance of $Z_{jmt}$
\end{enumerate}

\bigskip

\textbf{Immediate --- estimation}:
\begin{enumerate}
    \item Run sector first stage with updated grouped sectors (10 groups)
    \item Compare sector-level $F$-statistics across FE structures (sector$\times$year, muni$\times$year, year-only)
    \item Sector second stage: reduced form and vector 2SLS
\end{enumerate}

\bigskip

\textbf{Design decisions to resolve}:
\begin{enumerate}
    \item Should the president tier be included in the sector instrument? (negative sign, crowd-out concern)
    \item Primary FE for sector first stage: sector$\times$year (most demanding) vs.\ muni$\times$year (strongest $F$ in previous results)
    \item If aligned firms are diffuse across sectors (low HHI), can we redesign the instrument to exploit cross-municipality variation instead?
\end{enumerate}

\end{frame}

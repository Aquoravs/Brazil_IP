\begin{frame}{Research Question}

\textbf{Is the sector-level allocation of BNDES lending across municipalities GDP-optimal?}

\bigskip

\begin{itemize}
    \item BNDES disburses $\sim$3\% of GDP annually in subsidized industrial lending
    \item Existing literature (Lane 2025, Juh\'asz 2018) shows IP ``works'' for targeted sectors
    \item \textbf{But}: does the policymaker target the \emph{right} sectors?
    \item We derive and test the \textbf{first-order condition for optimality}: at an interior optimum, marginal reallocations across sectors have zero effect on GDP
\end{itemize}

\bigskip

\textbf{Contribution}: Move from ``does IP work?'' to ``is IP \emph{optimal}?''
--- and if not, identify which sectors are over/under-supported

\end{frame}

% ===========================================================================
\begin{frame}{Theory: Local Optimality Test}

Sector shares $s_m \in \Delta^{J-1}$, municipal output $Y_m = Y_m(s_m)$.

\medskip

\textbf{Directional derivative condition} --- at any interior optimum:
\begin{equation*}
    \frac{dY_m}{d\varepsilon}\bigg|_{\varepsilon=0}
    = \nabla_{s_m} Y_m \cdot v_m = 0
    \quad\text{for all } v_m \perp \iotaVec
\end{equation*}

\textbf{Empirical projection} (within-simplex, $\iotaVec'\Delta s_{mt}=0$):
\begin{equation*}
    \Delta Y_{m,t+h} = \beta' \Delta s_{mt} + u_{m,t+h}
\end{equation*}

\textbf{Null hypothesis (optimality)}:
\begin{equation*}
    H_0: \quad \beta = \mathbf{0}
\end{equation*}

\begin{itemize}
    \item Model-free: valid under any smooth objective
    \item $\hat\beta_j > 0$: sector $j$ under-supported $\implies$ tilt more toward $j$
    \item $\hat\beta_j < 0$: sector $j$ over-supported
\end{itemize}

\end{frame}

% ===========================================================================
\begin{frame}[allowframebreaks]{Identification: Shift-Share IV from Political Alignment}

$\Delta s_{mt}$ is endogenous $\implies$ need instruments that shift the sector mix exogenously.

\bigskip

\textbf{Shift} --- Political alignment turnover:
\begin{itemize}
    \item When party $p$ wins office $\ell \in \{\text{mayor, governor, president}\}$, BNDES credit tilts toward $p$-affiliated firms (Lazzarini et al.~2015, Carvalho 2014)
    \item $\Delta\Align^\ell_{mpt}$: party-$p$ alignment shock in municipality $m$, year $t$
\end{itemize}

\medskip

\textbf{Share} --- Baseline sectoral exposure to each party:
\begin{equation*}
    w_{jmp,\tau} \equiv \frac{\sum_{f \in \mathcal{F}(j,m)} L_{f,p,\tau}}
    {\sum_{f \in \mathcal{F}(j,m)} L_{f,\tau}}
\end{equation*}
where $L_{fp}$ = number of owners of firm $f$ affiliated to party $p$; $L_f$ = total owners (affiliated and unaffiliated).

\textbf{Instrument} (sector-specific, within municipality):
\begin{equation*}
    Z^\ell_{jmt} \equiv \sum_p w_{jmp,\tau} \cdot \Delta\Align^\ell_{mpt}
\end{equation*}

\end{frame}

% ===========================================================================
\begin{frame}{Estimation Strategy}

\textbf{Firm first stage} (firm $\times$ municipality $\times$ year):
\begin{equation*}
    \mathbf{1}(\text{BNDES}_{fmt} > 0) = \sum_\ell \lambda_\ell\, \FA^\ell_{fmt}
    + \gamma_f + \alpha_{mt} + u_{fmt}
\end{equation*}
\begin{itemize}
    \item $\FA^\ell_{fmt} = \sum_p \omega_{fp,\tau} \Align^\ell_{mpt}$: firm-level alignment exposure
    \item FE: firm + municipality$\times$year; clustering: firm + municipality; weights: employment
    \item Also estimated for intensive margin ($\log$ BNDES $|$ BNDES $> 0$) and in changes
\end{itemize}

\medskip

\textbf{Sector first stage} (municipality $\times$ sector $\times$ year):
\begin{equation*}
    \Delta s_{jmt} = \sum_\ell \lambda_\ell\, Z^\ell_{jmt} + \phi\,\EC_{jm,\tau} + \gamma_{jm} + \alpha_{jt} + u_{jmt}
\end{equation*}
\begin{itemize}
    \item $\EC_{jm,\tau}$: exposure control (overall political connectedness of sector-municipality cell)
    \item Primary FE: municipality$\times$sector + sector$\times$year; clustering: municipality + sector
    \item Identification from \textbf{cross-sector variation} in party exposure $w_{jmp,\tau}$
\end{itemize}

\end{frame}

\begin{frame}{Key Takeaways}

\textbf{Firm first stage}:
\begin{itemize}
    \item Political alignment predicts firm-level BNDES access and loan size
    \item Effect operates through both extensive and intensive margins
    \item Validates the micro political-lending channel underlying the sector-level instruments
\end{itemize}

\bigskip

\textbf{Sector first stage}:
\begin{itemize}
    \item Baseline (sector$\times$year FE): instruments are weak under coalition alignment
    \item Municipality$\times$year FE and year-only FE yield stronger first stages
    \item Party-level alignment produces distinct variation at subnational tiers
    \item Exposure control matters: coefficients drop without it
    \item 2002-fixed weights are weaker than cycle-specific across the board
\end{itemize}

\bigskip

{\color{alertred} Main question: which FE structure and alignment definition should be primary? Coalition with sector$\times$year FE is the most demanding but weakest specification.}

\end{frame}
